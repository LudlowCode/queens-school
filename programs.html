Program 3


# Imports go at the top
from microbit import *

#Use variables to record the names and pictures
nameA = 'Ada'
pictureA = Image.MUSIC_CROTCHET
nameB = 'Muhammad'
pictureB = Image.BUTTERFLY
default_image = Image.ASLEEP

# Code in a 'while True:' loop repeats forever
while True:
    #Show this for button A..
    if button_a.is_pressed():
        display.scroll(nameA)
        display.show(pictureA)
        sleep(1000)
    #or show this for button B..
    elif button_b.is_pressed():
        display.scroll(nameB)
        display.show(pictureB)
        sleep(1000)
    #or this when no button is presssed..
    else:
        display.show(default_image)
